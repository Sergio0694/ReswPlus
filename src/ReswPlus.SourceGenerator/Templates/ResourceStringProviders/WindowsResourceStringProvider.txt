using Windows.ApplicationModel.Resources;

namespace ReswPlus
{
    public class ResourceStringProvider
    {
        private readonly ResourceLoader _resourceLoader;

        public ResourceStringProvider(string resourceMapId)
        {
            _resourceLoader = ResourceLoader.GetForViewIndependentUse(resourceMapId);
        }

        public string GetString(string resourceId) => _resourceLoader.GetString(resourceId);
    }
}
